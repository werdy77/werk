 10 /**
 11  * @class Show messages in a popup window.
 12  * @param {string} pTargetId target HTML element's ID
 13  */
 14 function DebugMessages(pTargetId) {
 15 	Template.Save(DebugMessages._template);
 16 
 17 	DebugMessages._window = new PopupWindow(pTargetId);
 18 	DebugMessages._page = new Page();
 19 	DebugMessages._page.ShowTemplate(Template.Generate('debugmessages'),pTargetId);
 20 	DebugMessages._window.RegisterForMove('debugmessages_header');
 21 	
 22 	DebugMessages._log_node = document.getElementById('debugmessages_log');
 23 	
 24 	Messages.debug = true;
 25 	
 26 	Messages.Register({sender:'sent_messages',message:'WIDGET_VALUE_CHANGED',funct:DebugMessages._SetLoggingSentMessages});
 27 	Messages.Register({sender:'received_messages',message:'WIDGET_VALUE_CHANGED',funct:DebugMessages._SetLoggingReceivedMessages});
 28 	Messages.Register({sender:'log_messages',message:'WIDGET_VALUE_CHANGED',funct:DebugMessages._SetLoggingLogMessages});
 29 	Messages.Register({sender:'clear_log',message:'WIDGET_BUTTON_CLICKED',funct:DebugMessages.ClearLog});
 30 };
 31 	/**
 32 	 * Start/stop logging sent messages. @private
 33 	 */
 34 	DebugMessages._SetLoggingSentMessages = function() {
 35 		if(DebugMessages._page) {
 36 			if(DebugMessages._page.WidgetValue('sent_messages')) {
 37 				DebugMessages._ShowLog();
 38 				DebugMessages.sent_messages_reg = Messages.Register(
 39 					{sender:'Messages',message:'DEBUG_SEND_MESSAGE',funct:DebugMessages._Logging});
 40 			}
 41 			else {
 42 				Messages.Unregister(DebugMessages.sent_messages_reg);
 43 				DebugMessages._HideLogIfAllDisabled();
 44 			}
 45 		}
 46 	};
 47 
 48 	/**
 49 	 * Start/stop logging received messages. @private
 50 	 */
 51 	DebugMessages._SetLoggingReceivedMessages = function() {
 52 		if(DebugMessages._page) {
 53 			if(DebugMessages._page.WidgetValue('received_messages')) {
 54 				DebugMessages._ShowLog();
 55 				DebugMessages.received_messages_reg = Messages.Register(
 56 					{sender:'Messages',message:'DEBUG_MESSAGE_RECEIVED',funct:DebugMessages._Logging});
 57 			}
 58 			else {
 59 				Messages.Unregister(DebugMessages.received_messages_reg);
 60 				DebugMessages._HideLogIfAllDisabled();
 61 			}
 62 		}
 63 	};
 64 
 65 	/**
 66 	 * Start/stop logging log messages. @private
 67 	 */
 68 	DebugMessages._SetLoggingLogMessages = function() {
 69 		if(DebugMessages._page) {
 70 			if(DebugMessages._page.WidgetValue('log_messages')) {
 71 				DebugMessages._ShowLog();
 72 			}
 73 			else {
 74 				DebugMessages._HideLogIfAllDisabled();
 75 			}
 76 		}
 77 	};
 78 
 79 	/**
 80 	 * Set logging for sent/received messages.
 81 	 * @param {boolean} pForSentMesssages for sent messages
 82 	 * @param {boolean} pForReceivedMesssages for received messages
 83 	 */
 84 	DebugMessages.SetLogging = function(pForSentMessages, pForReceivedMessages, pForLogMessages) {
 85 		if(DebugMessages._page) {
 86 			DebugMessages._page.WidgetValue('sent_messages', pForSentMessages);
 87 			DebugMessages._page.WidgetValue('received_messages', pForReceivedMessages);
 88 			DebugMessages._page.WidgetValue('log_messages', pForLogMessages);
 89 		}
 90 	};
 91 
 92 	/**
 93 	 * Show log section. @private
 94 	 */
 95 	DebugMessages._ShowLog = function() {
 96 		if(DebugMessages._log_node) DebugMessages._log_node.style.height = '200px';
 97 	};
 98 
 99 	/**
100 	 * Hide log section, show only the header. @private
101 	 */
102 	DebugMessages._HideLog = function() {
103 		if(DebugMessages._log_node) DebugMessages._log_node.style.height = '0px';
104 	};
105 	
106 	/**
107 	 * Hide log section, show only the header. @private
108 	 */
109 	DebugMessages._HideLogIfAllDisabled = function() {
110 		if(!DebugMessages._page.WidgetValue('sent_messages') && !DebugMessages._page.WidgetValue('received_messages') && !DebugMessages._page.WidgetValue('log_messages')) DebugMessages._HideLog();
111 	};
112 
113 	/**
114 	 * Clear log.
115 	 */
116 	DebugMessages.ClearLog = function() {
117 		DebugMessages._log = '';
118 		DebugMessages._DisplayLog('');
119 	};
120 
121 	/**
122 	 * Show user interface (window).
123 	 */
124 	DebugMessages.ShowUI = function() {
125 		if(DebugMessages._window) DebugMessages._window.Show();
126 	};
127 
128 	/**
129 	 * Hide user interface (window).
130 	 */
131 	DebugMessages.HideUI = function() {
132 		if(DebugMessages._window) DebugMessages._window.Hide();		
133 	};
134 	
135 	/**
136 	 * Show message in log field. @private
137 	 * @param {string} pMessage message to display in log
138 	 */
139 	DebugMessages.Log = function(pMessage) {
140 		if(DebugMessages._page && DebugMessages._page.WidgetValue('log_messages')) {
141 			var now = new Date();
142 			
143 			DebugMessages._log = '<b>L '+now.getHours()+':'+now.getMinutes()
144 				+':'+now.getSeconds()+'.'+now.getMilliseconds()+'</b> '
145 				+pMessage+'<br/>'
146 				+DebugMessages._log;
147 			DebugMessages._DisplayLog(DebugMessages._log);
148 		}
149 	};
150 	
151 	/**
152 	 * Show message in log field. @private
153 	 * @param {string} params.sender message's owner
154 	 * @param {string} params.message (eg. JS_LOADED)
155 	 * @param {function} params.funct target function
156 	 */
157 	DebugMessages._Logging = function(params) {
158 		if(DebugMessages._page) {
159 			var now = new Date();
160 			var method = (params.message == 'DEBUG_SEND_MESSAGE') ? 'S' : 'R';
161 			
162 			DebugMessages._log = '<b>'+method+' '+now.getHours()+':'+now.getMinutes()
163 				+':'+now.getSeconds()+'.'+now.getMilliseconds()+'</b> '
164 				+params.userdata.sender+' ⇒ '+params.userdata.message+'<br/>'
165 				+DebugMessages._log;
166 			DebugMessages._DisplayLog(DebugMessages._log);
167 		}
168 	};
169 
170 	/**
171 	 * Display log (string) in the window. @private
172 	 * @param {string} pContent content
173 	 */
174 	DebugMessages._DisplayLog = function(pContent) {
175 		if(DebugMessages._log_node) DebugMessages._log_node.innerHTML = pContent;
176 	};

182 	/** storage for the page @private @type Page */
183 	DebugMessages._page = null;
184 	/** storage for the popup window @private @type PopupWindow */
185 	DebugMessages._window = null;
186 	/** sent messages receiver ID @private @type object */
187 	DebugMessages._sent_message_reg = null;
188 	/** received messages receiver ID @private @type object */
189 	DebugMessages._received_message_reg = null;

192 		name: 'debugmessages',
193 		html: '\

195  <div id="debugmessages_header" style="background:#bbb;padding:2px;">\

208 		}

211 /**
212  * @class Debug functions.
213  */
214 function Debug() {};
215 	/**
216 	 * Dump an object into a string.
217 	 * @param {string} pObjName object name
218 	 * @param {object} pObj object
219 	 * @param {boolean} pHtmlOutput if true formatted html output will be generated
220 	 * @return {string} dump
221 	 */
222 	Debug.DumpObject = function(pObjName, pObj, pHtmlOutput, pLevel) {
223 		if(pLevel == 0) return ''; // only 5 level deep dumped
224 		if(!pLevel) pLevel = 5;
225 		
226 		var dump = '';
227 		var obj_sep = '.';
228 		var key_value_sep = ' -> ';
229 		var new_line = '';
230 
231 		if(pHtmlOutput) {
232 			obj_sep = ' • ';
233 			key_value_sep = ' ⇒ ';
234 			new_line = '<br/>';
235 		}
236 
237 		for(var key in pObj) {
238 			if(typeof(pObj[key]) == 'object') {
239 				dump += pObjName+obj_sep+key+key_value_sep+"--- OBJECT ---"+new_line+"\n";
240 				dump += this.DumpObject(pObjName+obj_sep+key, pObj[key], pHtmlOutput, pLevel-1);
241 			}
242 			else dump += pObjName+obj_sep+key+key_value_sep+pObj[key]+new_line+"\n";
243 		}
244 		
245 		return dump;
246 	};

251 	 */
252 	Debug.StartBenchmark = function() {
253 		Debug._benchmark_started_at = (new Date()).getTime();
254 	};

257 	 * Return time in millisec ellapsed since {@link #StartBenchmark} call.
258 	 * @return {integer} ellapsed time in millisec
259 	 */
260 	Debug.StopBenchmark = function() {
261 		return (new Date()).getTime() - Debug._benchmark_started_at;
262 	};
263
