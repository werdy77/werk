201 	/**
202 	 * Register a widget for an existing html element.
203 	 * @param {object} pReg register information {id, name, value, filter, valid, required, disabled}
204 	 * @return (Widget|null) created widget or null on error
205 	 */
206 	Widget.Register = function(pReg) {
207 		var widget = null;
208 
209 		// create new input by type
210 		var node = document.getElementById(pReg.id);
211 		
212 		// create widget
213 		// choose type by type parameter OR node type
214 		if(pReg.type == null) pReg.type = Widget._GetWidgetTypeByNode(node);
215 		switch(pReg.type) {
216 			case 'text': widget = new TextWidget(); break;
217 			case 'textarea': widget = new TextAreaWidget(); break;
218 			case 'password': widget = new PasswordWidget(); break;
219 			case 'checkbox': widget = new CheckboxWidget(); break;
220 			case 'hidden': widget = new HiddenWidget(); break;
221 			case 'radio': widget = new RadioWidget(); break;
222 			case 'select': widget = new SelectWidget(); break;
223 			case 'button': widget = new ButtonWidget(); break;
224 			case 'html': widget = new HtmlWidget(); break;
225 			case 'image': widget = new ImageWidget(); break;
226 			case 'slider': widget = new SliderWidget(); break;
227 		}
228 		
229 		// if widget is valid set options
230 		if(widget) {
231 			widget.Init(node, pReg.name);
232 			if(pReg.filter && ContentFilter[pReg.filter]) widget.SetContentFilter(ContentFilter[pReg.filter]);
233 			if(pReg.valid && ContentValidator[pReg.valid]) widget.SetContentValidator(ContentValidator[pReg.valid]);
234 			if(pReg.value != null) widget.Value(pReg.value);
235 			if(pReg.required) widget.Required(true);
236 			widget.Disabled(pReg.disabled);
237 		}
238 		
239 		return widget;
240 	};

551 	 * Read/set cursor position.
552 	 * @param {integer|null} pStartPos selection start position
553 	 * @param {integer|null} pEndPos selection end position
554 	 * @param {object} cursor position {start, end}
555 	 */
556 	TextWidget.prototype.CursorPosition = function(pStartPos, pEndPos) {
557 		var pos = {start:0,end:0};
558 		
559 		if(this._node && this._focused) {
560 			// set
561 			if(pStartPos != null) {
562 				if(pEndPos == null) pEndPos = pStartPos;
563 				
564 				// standard way
565 				if(this._node.selectionStart != null) {
566 					this._node.setSelectionRange(pStartPos, pEndPos);
567 					
568 					if(globals.browser.Gecko) {
569 						// Trigger a "space" keypress.
570 						var evt = document.createEvent("KeyboardEvent");
571 						evt.initKeyEvent("keypress", true, true, null, false, false, false, false, 0, 32);
572 						this._node.dispatchEvent(evt);
573 
574 						// Trigger a "backspace" keypress.
575 						evt = document.createEvent("KeyboardEvent");
576 						evt.initKeyEvent("keypress", true, true, null, false, false, false, false, 8, 0);
577 						this._node.dispatchEvent(evt);
578 					}
579 				}
580 		
581 				// ie way (the very bad way)
582 				else if(document.selection != null) {
583 					var range = this._node.createTextRange();
584 					range.collapse(true);
585 					range.moveEnd('character', pEndPos);
586 					range.moveStart('character', pStartPos);
587 					range.select();
588 				}
589 				
590 				pos.start = pStartPos;
591 				pos.end = pEndPos;
592 			}
593 			
594 			// get
595 			else {		
596 				// gecko way
597 				if(this._node.selectionStart != null) {
598 					pos.start = this._node.selectionStart;
599 					pos.end = this._node.selectionEnd;
600 				}
601 	
602 				// ie way (the very bad way)
603 				else if(document.selection != null) {
604 					var range = document.selection.createRange();
605 					var range2 = this._node.createTextRange();
606 					range2.setEndPoint('EndToStart', range);
607 					pos.start = range2.text.length;
608 					pos.end = pos.start+range.text.length;
609 				}
610 			}
611 		}
612 		
613 		return pos;
614 	};

642 	/**
643 	 * Handle key pressed events and do character filtering. @private
644 	 * @param {Event} pEvent keypressed event
645 	 * @return {boolean} false, if character isn't allowed
646 	 */
647 	TextWidget.prototype._FilterChar = function(pEvent) {
648 		// send WIDGET_ENTER_PRESSED message
649 		if(pEvent.keyCode == 13) Messages.Send({sender:this._name, message:'WIDGET_ENTER_PRESSED'});
650 		
651 		/// allow some control keys
652 		if(
653 			pEvent.keyCode < 32 || // control char
654 			pEvent.keyCode == 63272 || // delete - safari
655 			(pEvent.keyCode >= 37 && pEvent.keyCode <= 40) || // arrows - opera
656 			(pEvent.keyCode >= 63232 && pEvent.keyCode <= 63235) || // arrows - safari
657 			(pEvent.ctrlKey && pEvent.keyCode == 99) || // ctrl+c
658 			(pEvent.metaKey && pEvent.keyCode == 99) || // apple+c on mac
