189 	 * Give back number of days in the given year/month.
190 	 * @param (number) params.year year
191 	 * @param (number) params.month month
192 	 * @return (number) day's number
193 	 */
194 	Misc.GetDaysPerMonth = function(params)
195 	{
196 		var days_per_month = [31,28,31,30,31,30,31,31,30,31,30,31];
197 		var month = params.month.valueOf()-1;
198 		var year = params.year.valueOf();
199 		var days = days_per_month[month];
200 	
201 		// leap year count
202 		if(
203 			month == 1 &&
204 			(year || year.length)
205 		) {
206 			if(
207 				(year%400 == 0) || 
208 				((year%4 == 0) && (year%100 != 0))
209 			) days++;
210 		}
211 
212 		return days;
213 	};

215 	/*
216 	 * Give back number of days in the given year/month.
217 	 * @param (number) params.year year
218 	 * @param (number) params.month month
219 	 * @param (number) params.day day
220 	 * @param (number) params.no_past_date past date not allowed - optional
221 	 */
222 	Misc.IsValidDate = function(params) {
223 		// if not all filled, no error
224 		if(
225 			params.year.length == 0 || 
226 			params.day.length == 0 || 
227 			params.month.length == 0
228 		) return false;
229 
230 		var year = params.year.valueOf();
231 		var month = params.month.valueOf();
232 		var day = params.day.valueOf();
233 
234 		// check 'no past date' option
235 		if(params.no_past_date) {
236 			var date = new Date(year, month-1, day);
237 			var now = new Date();
238 			now.setHours(0,0,0,0);
239 			if(date.getTime() < now.getTime()) return false;
240 		}
241 
242 		// check valid month
243 		if(month > 12 || month < 1) return false;
244 
245 		// check valid day
246 		if(day > Misc.GetDaysPerMonth(params) || day < 1) return false;
247 	
248 		return true;
249 	};

273 /** 
274  * @class Handle timed and/or repeated async calls.
275  * @param {object} pObj called object
276  * @param {function} pFunct called function's name (of given object if its exist)
277  * @param {any} pArg1... parameters
278  */
279 function AsyncCall(pObj, pFunct, pArg1, pArg2, pArg3, pArg4, pArg5) {
280 	/** called object @type object */
281 	this.obj = pObj;
282 	/** called function @type function */
283 	this.funct = pFunct;
284 	
285 	/** parameter @type any */
286 	this.arg1 = pArg1;
287 	/** parameter @type any */
288 	this.arg2 = pArg2;
289 	/** parameter @type any */
290 	this.arg3 = pArg3;
291 	/** parameter @type any */
292 	this.arg4 = pArg4;
293 	/** parameter @type any */
294 	this.arg5 = pArg5;
295 	
296 	/** generated ID @private @type integer */
297 	this._id = Misc.GenerateUID();
298 
299 	AsyncCall._calls[this._id] = this;	
300 };
301 	/**
302 	 * Execute async call.
303 	 * @param {integer} pDelay delay in ms
304 	 * @param {boolean} pRepeat if true target will called in every pDelay ms until {@link #Stop}
305 	 * @return {boolean} false if call has already executed
306 	 */
307 	AsyncCall.prototype.Execute = function(pDelay, pRepeat) {
308 		if(this.timerId) return false; // its already run
309 		if(!pDelay) pDelay = 0;
310 		
311 		if(pRepeat) {
312 			this.timerId = setInterval('AsyncCall._AsyncCallback('+this._id+');', pDelay);
313 		} else {
314 			setTimeout('AsyncCall._AsyncCallback('+this._id+');', pDelay);
315 		}
316 		
317 		return true;
318 	};
319 
320 	/**
321 	 * Stop a runing repeated (loop) call.
322 	 */
323 	AsyncCall.prototype.Stop = function() {
324 		if(this.timerId) {
325 			clearInterval(this.timerId);
326 			this.timerId = null;
327 			this.Destroy();
328 		}
329 	};
330 	
331 	/**
332 	 * Dereference from {@link #_calls}.
333 	 */
334 	AsyncCall.prototype.Destroy = function() {
335 		delete AsyncCall._calls[this._id];
336 	};
337 	
338 	/**
339 	 * The real JS callback (must be global). @private
340 	 * @param {integer} pId ID of an AsyncCall object
341 	 */
342 	AsyncCall._AsyncCallback = function(pId) {
343 		var self = AsyncCall._calls[pId];
344 		self.funct.call(self.obj, self.arg1, self.arg2, self.arg3, self.arg4, self.arg5);
345 		if(!self.timerId) self.Destroy(); // remove reference 
346 	};

354  * Search item by value @addon
355  * @param {any} pValue value
356  * @param {integer} pStartIndex start search from
357  * @return {integer} index of searched value or -1 if not found
358  */
359 Array.prototype.searchItem = function(pValue, pStartIndex) {
360 	if(pStartIndex == null) pStartIndex = 0;
361 	
362 	for(var i=pStartIndex; i<this.length; i++) {
363 		if(this[i] == pValue) {
364 			return i;
365 			break;
366 		}
367 	}
368 	
369 	return -1;
370 };

372 /**
373  * Delete an item from the array by value or index. Extension for Array class. @addon
374  * @param {any|null} pValue deletable item or null if index has given
375  * @param {integer} pIndex deletable item's index (faster)
376  * @return {boolean} true if item found and deleted
377  */
378 Array.prototype.deleteItem = function(pValue, pIndex) {
379 	var status = false;
380 	
381 	// search pIndex
382 	if(pIndex == null) pIndex = this.searchItem(pValue);
383 	
384 	if(-1 < pIndex && pIndex < this.length ) {
385 		// remove item
386 		var length = this.length-1;
387 		while(pIndex < length) this[pIndex] = this[++pIndex];
388 		this.pop(); // last entry is duplicated
389 		status = true;
390 	}
391 	
392 	return status;
393 };

448 /**
449  * Convert date to string what SQL understand.
450  * @param {boolean} pToUTC true if convert to UTC date string
451  * @param {boolean} pWithSeconds true to include seconds too
452  * @return {string} result SQL date string (like "2014-02-20 06:14:47")
453  */
454 Date.prototype.toSQLDate = function(pToUTC, pWithSeconds) {
455 	var date = this;
456 	
457 	if(pWithSeconds == null) pWithSeconds = true;
458 	
459 	// convert local time to UTC
460 	if(pToUTC) {
461 		date = new Date();
462 		date.setTime(this.getTime()+(date.getTimezoneOffset()*60000));
463 	}
464 	
465 	return date.getFullYear()+'-'+String.withTrailingZeros((date.getMonth()+1),2)+'-'+String.withTrailingZeros(date.getDate(),2)
466 		+' '+String.withTrailingZeros(date.getHours(),2)+':'+String.withTrailingZeros(date.getMinutes(),2)
467 		+(pWithSeconds ? ':'+String.withTrailingZeros(date.getSeconds(),2) : '');
468 };

471  * Convert SQL date string to JS date object.
472  * @param {string} pSqlDate date in SQL date string (like "2014-02-20 06:14:47")
473  * @param {boolean} pIsUTC true if SQL date is in UTC
474  * @return {object} JS date
475  */
476 Date.parseSQLDate = function(pSqlDate, pIsUTC) {
477 	if(pSqlDate) {
478 		var dateParts = pSqlDate.split(RegExp.Cache('[- :]'));
479 		if(dateParts.length >= 5) {
480 			var date = new Date(dateParts[0], dateParts[1]-1, dateParts[2], dateParts[3], dateParts[4], (dateParts.length > 5 ? dateParts[5] : null));
481 			
482 			// convert UTC time to local
483 			if(pIsUTC) date.setTime(date.getTime()-(date.getTimezoneOffset()*60000));
484 			
485 			return date;
486 		}
487 	}
488 	return null;
489 };

569  * @param {string} pName Lock's name (ID), eg. 'my_function'
570  * @param {object} pObj called object (pFunct parent)
571  * @param {array} pArgs arguments in an array (eg. function.arguments)

574 function Lock(pName, pObj, pArgs, pFunct) {
575 	if(Lock._locks[pName] == null) {
576 		Lock._locks[pName] = [];
577 	
578 		// call function
579 		pFunct.callWithArguments(pObj, pArgs);
580 		
581 		// call saved functions
582 		while(Lock._locks[pName].length) {
583 			// create call
584 			var o = Lock._locks[pName][0];

588 		}
589 
590 		delete Lock._locks[pName];
591 	}

593 	// save function for later exec
594 	else {
595 		Lock._locks[pName].push({
596 			obj: pObj,
597 			funct: pFunct,
598 			args: pArgs
599 		});
600 	}

607 /**
608  * @class Handle async functioning via messages. 
609  * Messages are simple strings like 'JS_LOADED' or 'WIDGET_BUTTON_CLICKED'.
610  */
611 function Messages() {};
612 	/**
613 	 * Register a function to receive a message.
614 	 * RegExp pattern can be used in the sender/message string.
615 	 * @param {string} params.sender message's owner
616 	 * @param {string} params.message (eg. JS_LOADED)
617 	 * @param {function} params.funct receiver function
618 	 * @param {object} params.obj receiver object, optional
619 	 * @param {boolean} params.once receive only one message (and unregister automatically), optional
620 	 * @return {integer} result.uid unique ID of the receiver
621 	 */
622 	Messages.Register = function(params) {
623 		var result = {};
624 		
625 		if(params.funct != null && params.sender != null && params.message != null) {
626 			// store receiver
627 			result.uid = Misc.GenerateUID();
628 			Messages._receivers[result.uid] = params;
629 		
630 			// search permanent messages
631 			for(var id in Messages._messages) {
632 				if(
633 					RegExp.Cache('^'+params.sender+'$').test(Messages._messages[id].sender) &&
634 					RegExp.Cache('^'+params.message+'$').test(Messages._messages[id].message)
635 				) {
636 					// send message
637 					Messages._messages[id].receiver_uid = result.uid;
638 					var call = new AsyncCall(null, Messages._SendReal, Messages._messages[id]);
639 					call.Execute();
640 				}
641 			}
642 		}
643 		
644 		return result;
645 	};
646 
647 	/**
648 	 * Unregister a function from receiving further message.
649 	 * @param {integer} params.uid unique ID of receiver (given back at register)
650 	 */
651 	Messages.Unregister = function(params) {
652 		if(params.uid && Messages._receivers[params.uid]) delete Messages._receivers[params.uid];
653 	};

655 	/**
656 	 * Send a message to the registered receivers.
657 	 * Permanent (stored) messages are sended at once when a receiver registers to them.
658 	 * @param {string} params.sender message's owner
659 	 * @param {string} params.message (eg. JS_LOADED)
660 	 * @param {boolean} params.permanent send message permanently, optional
661 	 * @param {object} params.userdata further data will be passed to the receiver, optional
662 	 * @return {integer} result.uid unique ID of stored message, optional
663 	 */
664 	Messages.Send = function(params) {
665 		var result = {};
666 		
667 		if(params.sender != null && params.message != null) {
668 			// store message
669 			if(params.permanent) {
670 				result.uid = Misc.GenerateUID();
671 				Messages._messages[result.uid] = params;
672 			}
673 
674 			// debug
675 			if(Messages.debug && params.sender != 'Messages') {
676 				Messages.Send({sender:'Messages', message:'DEBUG_SEND_MESSAGE', userdata:params});
677 			}
678 		
679 			// send message on another thread
680 			var call = new AsyncCall(null, Messages._SearchReceivers, params);
681 			call.Execute();
682 		}
683 		
684 		return result;
685 	};

743  * @class Handle standard js events in a better way.
744  */
745 function Event() {};
746 	/**
747 	 * Register a function to receive callback on an event.
748 	 * @param {Element} pElement source of event
749 	 * @param {string} pEventName event name without 'on', eg. 'keyup'
750 	 * @param {object|null} pReceiver receiver object
751 	 * @param {function} pFunct receiver function
752 	 */
753 	Event.Register = function(pElement, pEventName, pObj, pFunctName) {
754 		Misc.TouchObject(pElement, '_receivers');
755 		if(!pElement._receivers[pEventName]) {
756 			Misc.TouchObject(pElement._receivers, pEventName);
757 
758 			// register to event
759 			// IE, Opera
760 			if(window.attachEvent) {
761 				pElement.attachEvent('on'+pEventName, Event._OnEvent);
762 			}
763 			// Mozilla (gecko) and Safari
764 			else if(pElement.addEventListener) {
765 				pElement.addEventListener(pEventName, Event._OnEvent, false);
766 //				pElement['on'+pEventName] = Event._OnEvent; // prevent selection on window dragging
767 			}
768 		}
769 
770 		// save receiver
771 		var uid = Misc.GenerateUID();
772 		pElement._receivers[pEventName][uid] = {
773 			obj: pObj,
774 			funct: pFunctName,
775 			srcObj: pElement
776 		}
777 	};
778 	
779 	/**
780 	 * Handle events and search registered functions to receive callbacks. @private
781 	 * @param {Event} pEvent event (except IE)
782 	 */
783 	Event._OnEvent = function(pEvent) {
784 		if(window.event) pEvent = window.event;
785 		var obj = this;
786 		if(pEvent.srcElement != null) obj = pEvent.srcElement; // IE fix
787 
788 		while(obj) {
789 			if(obj._receivers != null) {
790 				var rcv = null;
791 				for(var uid in obj._receivers[pEvent.type]) {
792 					rcv = obj._receivers[pEvent.type][uid];
793 					
794 					// if function return false, stop search receivers (break events chain)
795 					if(rcv.srcObj == obj && rcv.funct && rcv.funct.call(rcv.obj, pEvent) == false) return false;
796 				}
797 			}
798 			obj = obj.parentNode;
799 		}
800 		return true;
801 	};

803 	/**
804 	 * Unregister a function to receive callback on an event.
805 	 * @param {Element} pElement source of event
806 	 * @param {string} pEventName event name without 'on', eg. 'keyup'
807 	 * @param {object|null} pReceiver receiver object
808 	 * @param {function} pFunct receiver function
809 	 */ 
810 	Event.Unregister = function(pElement, pEventName, pObj, pFunct) {
811 		if(pElement._receivers && pElement._receivers[pEventName]) {
812 			var rcv = null;
813 			for(var uid in pElement._receivers[pEventName]) {
814 				rcv = pElement._receivers[pEventName][uid];
815 				if(rcv.obj == pObj && rcv.funct == pFunct) delete pElement._receivers[pEventName][uid];
816 			}
817 			Misc.DeleteObjectIfEmpty(pElement._receivers, pEventName);
818 			
819 			// unregister from event if no receivers
820 			if(!pElement._receivers[pEventName]) {
821 				if(window.detachEvent) {
822 					pElement.detachEvent('on'+pEventName, Event._OnEvent);
823 				}
824 				// Mozilla (gecko) and Safari
825 				else if(pElement.removeEventListener) {
826 					pElement.removeEventListener(pEventName, Event._OnEvent, false);
827 				}				
828 			}
829 		}
830 	};

845 	/**
846 	 * Include (load) an external JS file.
847 	 * Attention! Identical js filenames with different paths could cause problems!
848 	 * @param {string} src url to javascript
849 	 */
850 	Include.LoadJs = function(src, loaded) {	
851 		if(!Include._files[src]) {
852 			var element = null;
853 			element = document.createElement('script');
854 			element.type = 'text/javascript';
855 			element.charset = globals.charset;
856 			element.src = src+'?version='+globals.version;
857 			element.loaded = loaded;
858 
859 			// append it to the <head node
860 			var head = document.getElementsByTagName('head').item(0);
861 			if(head && element) {
862 				head.appendChild(element);				
863 				Include._files[src] = element;
864 			}
865 		}
866 	};

869 	 * Include (load) an external CSS file.
870 	 * @param {string} src url to the CSS file
871 	 */
872 	Include.LoadCSS = function(src) {	
873 		if(!Include._files[src]) {
874 			var element = null;
875 			element = document.createElement('link');
876 			element.type = 'text/css';
877 			element.media = 'all';
878 			element.rel = 'stylesheet';
879 			element.href = src+'?version='+globals.version;
880 			element.src = element.href;
881 			element.loaded = true;
882 
883 			// append it to the <head node
884 			var head = document.getElementsByTagName('head').item(0);
885 			if(head && element) {
886 				head.appendChild(element);				
887 				Include._files[src] = element;
888 			}
889 		}
