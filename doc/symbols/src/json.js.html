 12 /**
 13  * @class This class creates an JSON request and stores its data.
 14  * Data will send as a GET request and received as an included javascript file.
 15  * JSON_ANSWER_RECEIVED message will sent with the method as owner when answer has received.
 16  * Server answer must call {@link GLOBALS#_JSONCallback} function with the answer.
 17  * @param {string} pUrl server url, what will receives the request
 18  * @param {string|integer} pMethod method identifier, will be included in the request as method=xxx
 19  * @param {object} pData data (key/value pairs) what will send to the server
 20  * @param {object} pCache data, what will included in the answer message as a parameter
 21  */
 22 function JSONGetRequest(pUrl, pMethod, pData, pCache) {
 23 	/** Unique ID to identify request. @type integer */
 24 	this.id = Misc.GenerateUID(); // uniq ID of the request
 25 //	this.id = 1;
 26 	/** server url, what will receives the request @type string */
 27 	this.url = pUrl;
 28 	/** method identifier, will be included in the request as method=xxx @type string */
 29 	this.method = pMethod;
 30 	/** data (key/value pairs) what will send to the server @type object */
 31 	this.data = pData;
 32 	/** data, what will included in the answer message as a parameter @type object */
 33 	this.cache = pCache;
 34 	/** error code @type integer */
 35 	this.httpcode = 0;
 36 };
 37 	/**
 38 	 * Send request to the server.
 39 	 */
 40 	JSONGetRequest.prototype.Send = function() {
 41 		if(!this.element) {
 42 			var src = this.url+'?method='+this.method+'&rid='+this.id;
 43 			for(var key in this.data) src += '&'+key+'='+encodeURIComponent(this.data[key]);
 44 		
 45 			// create new DOM element
 46 			this.element = document.createElement('script');
 47 			this.element.type = "text/javascript";
 48 			this.element.src = src;
 49 			this.element.charset = 'utf-8';
 50 
 51 			// append it to the <head
 52 			var head = document.getElementsByTagName('head').item(0);
 53 			if(head) head.appendChild(this.element);
 54 		}
 55 	};

 57 /**
 58  * @class This class creates an JSON request and stores its data.
 59  * Data will send in a form as POST request, received as javascript data in a HTML file.
 60  * JSON_ANSWER_RECEIVED message will sent with the method as owner when answer has received.
 61  * Server answer must call {@link GLOBALS#_JSONCallback} function with the answer.
 62  * @param {object} pForm form to send data 
 63  * @param {string|integer} pMethod method identifier, will be included in the request as method=xxx
 64  * @param {object} pData data (key/value pairs) what will send to the server
 65  * @param {object} pCache data, what will included in the answer message as a parameter
 66  */
 67 function JSONPostRequest(pFormNode, pUrl, pMethod, pData, pCache) {
 68 	/** Unique ID to identify request. @type integer */
 69 	this.id = Misc.GenerateUID(); // uniq ID of the request
 70 //	this.id = 1;
 71 	/** form to send data @type object */
 72 	this.form = pFormNode;
 73 	/** server url, what will receives the request @type string */
 74 	this.url = pUrl;
 75 	/** method identifier, will be included in the request as method=xxx @type string */
 76 	this.method = pMethod;
 77 	/** data (key/value pairs) what will send to the server @type object */
 78 	this.data = pData;
 79 	/** data, what will included in the answer message as a parameter @type object */
 80 	this.cache = pCache;
 81 	/** error code @type integer */
 82 	this.httpcode = 0;
 83 };
 84 	/**
 85 	 * Send request to the server.
 86 	 */
 87 	JSONPostRequest.prototype.Send = function() {
 88 		if(this.form != null) {
 89       this.form.action = this.url+'?method='+this.method+'&rid='+this.id+'&post=1';
 90       for(var key in this.data) this._AddInputToForm(key,this.data[key]);
 91 			this.form.submit();
 92 			
 93 	    // remove old elements
 94 	    this._RemoveInputsFromForm();
 95 		}
 96 	};
 97 	
 98 	/**
 99 	 * Add a key/value as a hidden input element to the form.
100 	 * @param {string} pKey key
101 	 * @param {string} pValue value
102 	 */
103 	JSONPostRequest.prototype._AddInputToForm = function(pKey,pValue) {
104 	  if(this.form != null) {
105 	    // search for existing input
106   	  var i = 0;
107   	  for(i = 0; i < this.form.elements.length; i++) {
108   	    if(this.form.elements[i].name == pKey) break;
109   	  }
110 	  
111   	  if(this.form.elements.length == i) {
112   			var element = document.createElement('input');
113   			element.type = 'hidden';
114   			element.name = pKey;
115   			element.value = pValue;
116   			element.generated = true;
117   			this.form.appendChild(element);
118   		} else {
119   		  this.form.elements[i].value = pValue;
120   		}
121   	}
122 	};
123 
124   /**
125    * Remove generated (by _AddInputToForm) hidden input elements from the form.
126    */
127   JSONPostRequest.prototype._RemoveInputsFromForm = function() {
128     if(this.form != null) {
129   	  for(i = 0; i < this.form.elements.length; i++) {
130         if(this.form.elements[i].generated) {
131           this.form.removeChild(this.form.elements[i]);
132         }
133   	  }      
134     }
135   };

137 /**
138  * @class Handle and manage JSON communication via {@link JSONRequest} entities.
139  */
140 function JSONManager() {};
141 	/**
142 	 * Set/get default keys/values what will be included in every request.
143 	 * Set only if pKey and pValue both given.
144 	 * @param {string} pKey key
145 	 * @param {string|null} pValue value
146 	 * @return {string} value
147 	 */
148 	JSONManager.DefaultValue = function(pKey, pValue) {
149 		if(pKey != null && arguments.length > 1) {
150 			if(pValue == null) delete JSONManager._defaults[pKey];
151 			else JSONManager._defaults[pKey] = pValue;
152 		}
153 		return (JSONManager._defaults[pKey] != null) ? JSONManager._defaults[pKey] : '';
154 	};
155 	
156 	/**
157 	 * Send an JSON request to the server (GET).
158 	 * @param {string|integer} pMethod method identifier, will be included in the request as method=xxx
159 	 * @param {object} pData data (key/value pairs) what will send to the server
160 	 * @param {object} pCache data, what will included in the answer message as a parameter
161 	 * @return (integer) request ID
162 	 */
163 	JSONManager.Request = function(pMethod, pData, pCache) {
164 		if(pData == null) pData = {};
165 		
166 		// set default values
167 		Misc.CopyObject(JSONManager._defaults, pData, true, true);
168 			
169 		// send GET request
170 	  var req = new JSONGetRequest(JSONManager.url, pMethod, pData, pCache);
171 		JSONManager._requests[req.id] = req;
172 	  req.Send();
173 	  JSONManager._SetBusyStatus(true);
174 
175 		// remove old requests
176 		JSONManager.Clean();
177 		
178 		return req.id;
179 	};
180 	
181 	/**
182 	 * Send an JSON request to the server in form (POST).
183 	 * @param {string|integer} pMethod method identifier, will be included in the request as method=xxx
184 	 * @param {object} pData data (key/value pairs) what will send to the server
185 	 * @param {object} pCache data, what will included in the answer message as a parameter
186 	 * @return (integer) request ID. -1 on error.
187 	 */
188 	JSONManager.PostRequest = function(pMethod, pData, pCache) {
189 	  var id = -1;
190 	  
191 		if(JSONManager.form != null) {
192   		if(pData == null) pData = {};
193 		
194   		// set default values
195   		Misc.CopyObject(JSONManager._defaults, pData, true, true);
196 		
197 			// send POST request
198   	  var req = new JSONPostRequest(JSONManager.form, JSONManager.url, pMethod, pData, pCache);
199   		JSONManager._requests[req.id] = req;
200       JSONManager._SendNextPostRequest();
201 
202   		// remove old requests
203   		JSONManager.Clean();
204 		
205   		id = req.id;
206   	}
207   	
208   	return id;
209 	};
210 
211   /**
212    * POST requests must be serialized.
213    * This function send the next request only if the answer to the previous one has received.
214    */
215   JSONManager._SendNextPostRequest = function() {
216 	  if(JSONManager._form_busy == 0) {
217 	    var form_req = null;
218 	    
219 	    // search next not completed request
220   		for(var id in JSONManager._requests) {
221 			  form_req = JSONManager._requests[id];
222         if(!form_req.done && form_req.form != null) {
223           JSONManager._form_busy = id;
224           form_req.Send();
225           JSONManager._SetBusyStatus(true);
226           break;
227         }
228       }
229 	  }
230   };

232 	/**
233 	 * Resend requests received with pHttpCode HTTP code.
234 	 * @param {integer} pHttpCode HTTP code
235 	 */
236 	JSONManager.ResendByHttpCode = function(pHttpCode) {
237 		for(var id in JSONManager._errors[pHttpCode]) {
238 			var req = JSONManager._errors[pHttpCode][id];
239 			
240 			if(req.element != null) { // remove JS element from HEAD
241 			  req.element.parentNode.removeChild(req.element);
242 			  delete req.element;
243 			}
244 			
245 			req.done = false;
246 			
247 			// set default values
248 			Misc.CopyObject(JSONManager._defaults, req.data, true, false);
249 			
250 			req.Send();
251 			JSONManager._SetBusyStatus(true);
252 		}
253 	};
254 
255 	/**
256 	 *  Remove old (returned) requests.
257 	 */
258 	JSONManager.Clean = function() {
259 		var req = null;
260 		for(var id in JSONManager._requests) {
261 			req = JSONManager._requests[id];
262 			if(req.done && !req.httpcode) {
263 			  
264 			  if(req.element != null) { // remove JS element from HEAD
265 				  req.element.parentNode.removeChild(req.element);
266 				}
267 				delete JSONManager._requests[id];
268 			}
269 		}
270 	};

272 	/**
273 	 * Remove failed requests returned with pHttpCode HTTP code.
274 	 * @param {integer} pHttpCode HTTP code
275 	 */
276 	JSONManager.CleanByHttpCode = function(pHttpCode) {
277 		var req = null;
278 		for(var id in JSONManager._errors[pHttpCode]) {
279 			req = JSONManager._errors[pHttpCode][id];
280 			if(req.done) {
281 				req.httpcode = 0;
282 
283 				// remove from errors
284 				delete JSONManager._errors[pHttpCode][id];
285 				Misc.DeleteObjectIfEmpty(JSONManager._errors, pHttpCode);
286 			}
287 		}
288 	};

290 	/**
291 	 * Set busy status.
292 	 */
293 	JSONManager._SetBusyStatus = function(pStatus) {
294 		if(pStatus) {
295 			JSONManager._busy++;
296 			// turn on anim
297 			if(JSONManager._busy == 1) Messages.Send({sender:'JSONManager', message:'JSON_ACTIVATED'});
298 		}
299 		else if(JSONManager._busy > 0) {
300 			JSONManager._busy--;
301 			// turn off anim
302 			if(JSONManager._busy == 0) Messages.Send({sender:'JSONManager', message:'JSON_INACTIVATED'});
303 		}
304 	};

306 	/** server url, what will receives the GET request @type string */
307 	JSONManager.url = '';
308 	/** send data in form / POST @type object */
309 	JSONManager.form = null;
310 	/** FORM status (0 if no POST transaction). @private @type integer */
311 	JSONManager._form_busy = 0;
312 	/** Storage default values. @private @type object */
313 	JSONManager._defaults = {};
314 	/** Storage requests. @private @type object */
315 	JSONManager._requests = {};
316 	/** Storage for requests returned with error. @private @type object */
317 	JSONManager._errors = {};
318 	/** JSON status (0 if no active transaction). @private @type integer */
319 	JSONManager._busy = 0;

321 /**
322  * Called when an JSON answer comes from the server. 
323  * Send JSON_ANSWER_RECEIVED message.
324  * @param {integer} params.rid unique ID to identify this answer's request pair
325  * @param {integer} params.httpcode 200 if everything is ok
326  * @param {string} params.message
327  * @param {object} params.data server's answer
328  */
329 function _JSONCallback(params) {
330 	var req = JSONManager._requests[params.rid]; // search request
331 	JSONManager._SetBusyStatus(false);
332 	
333 	if(req) {
334 		if(req.cache) params.cache = req.cache;
335 		req.done = true;		
336 		req.httpcode = params.httpcode;
337 
338 		Messages.Send({sender:req.method, message:'JSON_ANSWER_RECEIVED', userdata:params});
339 
340 		// save failed request
341 		if(params.httpcode) {
342 			Misc.TouchObject(JSONManager._errors, params.httpcode);
343 			JSONManager._errors[params.httpcode][params.rid] = req;
344 		}
345 		
346 		// delete corrected request from _errors
347 		else if(req.httpcode && JSONManager._errors[req.httpcode][params.rid]) {
348 			delete JSONManager._errors[req.httpcode][params.rid];
349 			
350 			// send message if all requests with pHttpCode corrected
351 			if(Misc.DeleteObjectIfEmpty(JSONManager._errors, req.httpcode)) {
352 				Messages.Send({sender:req.httpcode, message:'JSON_ERRORS_CORRECTED', userdata:{httpcode:req.httpcode}});
353 			}
354 		}
355 
356 	  // send next form (POST) request
357 	  if(req.form != null) {
358 	    JSONManager._form_busy = 0;
359 	    JSONManager._SendNextPostRequest();
360 	  }
361 	} 
362 	
363 	// request not found, send an universal JSON_ANSWER_RECEIVED message
364 	else {
365 		Messages.Send({sender:'JSONManager', message:'JSON_ANSWER_RECEIVED', userdata:params});
366 	}
367 };
